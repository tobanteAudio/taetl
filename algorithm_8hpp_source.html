772  Compare comp)
773  -> etl::pair<ForwardIter, ForwardIter>
774 {

776  auto max = first;
777 
778  if (first == last || ++first == last) { return {min, max}; }
779 
780  if (comp(*first, *min)) { min = first; }

783  max = first;
