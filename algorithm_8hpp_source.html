350 template <class ForwardIt>
351 constexpr auto rotate(ForwardIt first, ForwardIt n_first, ForwardIt last) -> ForwardIt
352 {
353  if (first == n_first) { return last; }
354  if (n_first == last) { return first; }
355 
356  ForwardIt read = n_first;
357  ForwardIt write = first;
358  ForwardIt next_read = first; // read position for when "read" hits "last"
359 
360  while (read != last)
361  {
362  if (write == next_read)
363  {
364  next_read = read; // track where "first" went
365  }
366  etl::iter_swap(write++, read++);
367  }
368 
369  // rotate the remaining sequence into place
370  (rotate)(write, next_read, last);
371  return write;
372 }

379 template <class ForwardIt, class UnaryPredicate>
380 constexpr auto partition(ForwardIt first, ForwardIt last, UnaryPredicate p) -> ForwardIt
381 {
382  first = find_if_not(first, last, p);
383  if (first == last) { return first; }
384 
385  for (ForwardIt i = next(first); i != last; ++i)

391  }
392  }

546 template <class RandomIt, class Compare>
547 constexpr auto sort(RandomIt first, RandomIt last, Compare comp) -> void
548 {
549  auto const N = distance(first, last);
550  if (N <= 1) { return; }
551 
552  auto const pivot = *next(first, N / 2);
553  auto m1 = partition(first, last, [&](auto const& elem) { return comp(elem, pivot); });
554  auto m2 = partition(m1, last, [&](auto const& elem) { return !comp(pivot, elem); });
555 
556  sort(first, m1, comp);
557  sort(m2, last, comp);
558 }
