864 [[nodiscard]] constexpr auto minmax_element(ForwardIter first, ForwardIter last,
865  Compare comp)
866  -> etl::pair<ForwardIter, ForwardIter>
867 {
868  auto min = first;
869  auto max = first;
870 
871  if (first == last || ++first == last) { return {min, max}; }
872 
873  if (comp(*first, *min)) { min = first; }
874  else
875  {
876  max = first;
877  }
878
