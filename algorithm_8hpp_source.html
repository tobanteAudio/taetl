637 template <typename ForwardIter, typename Compare>
638 [[nodiscard]] constexpr auto minmax_element(ForwardIter first, ForwardIter last,
639  Compare comp)
640  -> etl::pair<ForwardIter, ForwardIter>
641 {
642  auto min = first;
643  auto max = first;
644 
645  if (first == last || ++first == last) { return {min, max}; }
646 
647  if (comp(*first, *min)) { min = first; }
648  else
649  {
650  max = first;
651  }
652 
653  while (++first != last)
654  {
655  auto i = first;
656  if (++first == last)
657  {
658  if (comp(*i, *min)) { min = i; }
659  else if (!(comp(*i, *max)))
660  {
661  max = i;
662  }
663  break;
664  }
665 
666  if (comp(*first, *i))
667  {
668  if (comp(*first, *min)) { min = first; }
669  if (!(comp(*i, *max))) { max = i; }
670  }
671  else
672  {
673  if (comp(*i, *min)) { min = i; }
674  if (!(comp(*first, *max))) { max = first; }
675  }
676  }
677 
678  return {min, max};
679 }
