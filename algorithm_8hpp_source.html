703 [[nodiscard]] constexpr auto minmax_element(ForwardIter first, ForwardIter last,
704  Compare comp)
705  -> etl::pair<ForwardIter, ForwardIter>
706 {
707  auto min = first;
708  auto max = first;
709 
710  if (first == last || ++first == last) { return {min, max}; }
711 
712  if (comp(*first, *min)) { min = first; }
713  else
714  {
715  max = first;
716  }
717 
718  while (++first != last)
719  {
720  auto i = first;
721  if (++first == last)
722  {
723  if (comp(*i, *min)) { min = i; }
724  else if (!(comp(*i, *max)))

726  max = i;
727  }
728  break;

731  if (comp(*first, *i))

734  if (!(comp(*i, *max))) { max = i; }
735  }
736  else
737  {
738  if (comp(*i, *min)) { min = i; }
