623  Compare comp)
624  -> etl::pair<ForwardIter, ForwardIter>
625 {
626  auto min = first;
627  auto max = first;
628 
629  if (first == last || ++first == last) { return {min, max}; }
630 
631  if (comp(*first, *min)) { min = first; }
632  else

634  max = first;
635  }
