375 template <class ForwardIt>
376 constexpr auto rotate(ForwardIt first, ForwardIt n_first, ForwardIt last) -> ForwardIt
377 {
378  if (first == n_first) { return last; }
379  if (n_first == last) { return first; }
380 
381  ForwardIt read = n_first;
382  ForwardIt write = first;
383  ForwardIt next_read = first; // read position for when "read" hits "last"
384 
385  while (read != last)
386  {
387  if (write == next_read)
388  {
389  next_read = read; // track where "first" went
390  }
391  etl::iter_swap(write++, read++);
392  }
393 
394  // rotate the remaining sequence into place
395  (rotate)(write, next_read, last);
396  return write;
397 }

404 template <class ForwardIt, class UnaryPredicate>
405 constexpr auto partition(ForwardIt first, ForwardIt last, UnaryPredicate p) -> ForwardIt
406 {
407  first = find_if_not(first, last, p);
408  if (first == last) { return first; }
409

429 {

571 template <class RandomIt, class Compare>
572 constexpr auto sort(RandomIt first, RandomIt last, Compare comp) -> void
573 {
574  auto const N = distance(first, last);
575  if (N <= 1) { return; }
576 
577  auto const pivot = *next(first, N / 2);
578  auto m1 = partition(first, last, [&](auto const& elem) { return comp(elem, pivot); });
579  auto m2 = partition(m1, last, [&](auto const& elem) { return !comp(pivot, elem); });
580 
581  sort(first, m1, comp);
582  sort(m2, last, comp);
583 }
