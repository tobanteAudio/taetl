286 template <class InputIt1, class InputIt2, class Compare>
287 [[nodiscard]] constexpr auto
288 lexicographical_compare(InputIt1 first1, InputIt1 last1, InputIt2 first2,
289  InputIt2 last2, Compare comp) -> bool
290 {
291  for (; (first1 != last1) && (first2 != last2); ++first1, (void)++first2)
292  {
293  if (comp(*first1, *first2)) { return true; }
294  if (comp(*first2, *first1)) { return false; }
295  }
296  return (first1 == last1) && (first2 != last2);
297 }
