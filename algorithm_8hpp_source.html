639 [[nodiscard]] constexpr auto minmax_element(ForwardIter first, ForwardIter last,
640  Compare comp)
641  -> etl::pair<ForwardIter, ForwardIter>
642 {
643  auto min = first;
644  auto max = first;
645 
646  if (first == last || ++first == last) { return {min, max}; }
647 
648  if (comp(*first, *min)) { min = first; }
649  else
650  {
651  max = first;
652  }
653 
654  while (++first != last)
655  {
656  auto i = first;
