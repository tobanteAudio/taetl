420 template <class ForwardIt>
421 constexpr auto rotate(ForwardIt first, ForwardIt n_first, ForwardIt last) -> ForwardIt
422 {
423  if (first == n_first) { return last; }
424  if (n_first == last) { return first; }
425 
426  auto read = n_first;
427  auto write = first;
428  auto next_read = first;
429 
430  while (read != last)
431  {
432  if (write == next_read) { next_read = read; }
433  etl::iter_swap(write++, read++);
434  }
435 
436  rotate(write, next_read, last);
437  return write;
438 }
