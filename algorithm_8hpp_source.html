407 template <class ForwardIt>
408 constexpr auto rotate(ForwardIt first, ForwardIt n_first, ForwardIt last) -> ForwardIt
409 {
410  if (first == n_first) { return last; }
411  if (n_first == last) { return first; }
412 
413  auto read = n_first;
414  auto write = first;
415  auto next_read = first;
416 
417  while (read != last)
418  {
419  if (write == next_read) { next_read = read; }
420  etl::iter_swap(write++, read++);
421  }
422 
423  rotate(write, next_read, last);
424  return write;
425 }
