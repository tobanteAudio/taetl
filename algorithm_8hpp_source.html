437 template <typename ForwardIt>
438 constexpr auto rotate(ForwardIt first, ForwardIt n_first, ForwardIt last) -> ForwardIt
439 {
440  if (first == n_first) { return last; }
441  if (n_first == last) { return first; }
442 
443  auto read = n_first;
444  auto write = first;
445  auto next_read = first;
446 
447  while (read != last)
448  {
449  if (write == next_read) { next_read = read; }
450  etl::iter_swap(write++, read++);
451  }
452 
453  rotate(write, next_read, last);
454  return write;
455 }
