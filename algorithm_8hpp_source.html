546 template <typename ForwardIt>
547 constexpr auto rotate(ForwardIt first, ForwardIt n_first, ForwardIt last) -> ForwardIt
548 {
549  if (first == n_first) { return last; }
550  if (n_first == last) { return first; }
551 
552  auto read = n_first;
553  auto write = first;
554  auto next_read = first;
555 
556  while (read != last)
557  {
558  if (write == next_read) { next_read = read; }
559  etl::iter_swap(write++, read++);
560  }
561 
562  rotate(write, next_read, last);
563  return write;
564 }
