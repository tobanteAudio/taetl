315 template <class ForwardIt>
316 constexpr auto rotate(ForwardIt first, ForwardIt n_first, ForwardIt last)
317  -> ForwardIt
318 {
319  if (first == n_first) { return last; }
320  if (n_first == last) { return first; }
321 
322  ForwardIt read = n_first;
323  ForwardIt write = first;
324  ForwardIt next_read = first; // read position for when "read" hits "last"
325 
326  while (read != last)
327  {
328  if (write == next_read)
329  {
330  next_read = read; // track where "first" went
331  }
332  etl::iter_swap(write++, read++);
333  }
334 
335  // rotate the remaining sequence into place
336  (rotate)(write, next_read, last);
337  return write;
338 }

346 template <class BidirIt, class UnaryPredicate>
347 constexpr auto stable_partition(BidirIt f, BidirIt l, UnaryPredicate p)
348  -> BidirIt
349 {
350  auto const n = l - f;
351  if (n == 0) { return f; }
352  if (n == 1) { return f + p(*f); }
353  auto const m = f + (n / 2);
354  return rotate(stable_partition(f, m, p), m, stable_partition(m, l, p));
355 }

364 template <class InputIt1, class InputIt2, class Compare>
365 [[nodiscard]] constexpr auto
366 lexicographical_compare(InputIt1 first1, InputIt1 last1, InputIt2 first2,
367  InputIt2 last2, Compare comp) -> bool
368 {
369  for (; (first1 != last1) && (first2 != last2); ++first1, (void)++first2)
370  {
371  if (comp(*first1, *first2)) { return true; }
372  if (comp(*first2, *first1)) { return false; }
373  }
374  return (first1 == last1) && (first2 != last2);
