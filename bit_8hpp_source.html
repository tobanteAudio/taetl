116 template <typename T>
117 [[nodiscard]] constexpr auto countl_zero(T x) noexcept
118  -> enable_if_t<detail::is_unsigned_integer<T>::value, int>
119 {
120  auto const total_bits = etl::numeric_limits<T>::digits;
121  if (x == T {0}) { return etl::numeric_limits<T>::digits; }
122 
123  int res = 0;
124  while (!(x & (T {1} << (total_bits - 1))))
125  {
126  x = (x << T {1});
127  res++;
128  }
129 
130  return res;
131 }
