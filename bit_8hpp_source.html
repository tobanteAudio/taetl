109  -> etl::enable_if_t<detail::is_unsigned_integer<T>::value, int>
110 {

147 template <typename T>
148 [[nodiscard]] constexpr auto countl_zero(T x) noexcept
149  -> enable_if_t<detail::is_unsigned_integer<T>::value, int>
150 {
151  auto const total_bits = etl::numeric_limits<T>::digits;
152  if (x == T {0}) { return etl::numeric_limits<T>::digits; }
153 
154  int res = 0;
155  while (!(x & (T {1} << (total_bits - 1))))
156  {
157  x = (x << T {1});
158  res++;
159  }
160 
161  return res;
162 }
