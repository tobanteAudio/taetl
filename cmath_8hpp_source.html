189 template <typename Float>
190 [[nodiscard]] constexpr auto lerp_impl(Float a, Float b, Float t) noexcept
191  -> ::etl::enable_if_t<::etl::is_floating_point_v<Float>, Float>
192 {
193  if ((a <= 0 && b >= 0) || (a >= 0 && b <= 0)) { return t * b + (1 - t) * a; }
194 
195  if (t == 1) { return b; }
196 
197  auto const x = a + t * (b - a);
198  if ((t > 1) == (b > a)) { return b < x ? x : b; }
199  return x < b ? x : b;
200 }
201 } // namespace detail
