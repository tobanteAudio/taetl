163 template <typename FloatT>
164 constexpr FloatT lerp_impl(FloatT a, FloatT b, FloatT t) noexcept
165 {
166  if ((a <= 0 && b >= 0) || (a >= 0 && b <= 0)) { return t * b + (1 - t) * a; }
167 
168  if (t == 1) { return b; }
169 
170  auto const x = a + t * (b - a);
171  if (t > 1 == b > a) { return b < x ? x : b; }
172  return x < b ? x : b;
173 }
174 } // namespace detail
