163 template <typename Float>
164 [[nodiscard]] constexpr auto lerp_impl(Float a, Float b, Float t) noexcept
165  -> ::etl::enable_if_t<::etl::is_floating_point_v<Float>, Float>
166 {
167  if ((a <= 0 && b >= 0) || (a >= 0 && b <= 0)) { return t * b + (1 - t) * a; }
168 
169  if (t == 1) { return b; }
170 
171  auto const x = a + t * (b - a);
172  if (t > 1 == b > a) { return b < x ? x : b; }
173  return x < b ? x : b;
174 }
175 } // namespace detail
