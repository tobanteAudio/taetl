45 auto iota(int val, char* const buffer, int base) -> char*
46 {
47  switch (base)
48  {
49  case 10:
50  {
51  auto digits10 = [](auto x) {
52  auto result = 1;
53  while (true)
54  {
55  if (x < 10) { return result; }
56  if (x < 100) { return result + 1; }
57  if (x < 1'000) { return result + 2; }
58  if (x < 10'000) { return result + 3; }
59 
60  x /= 10'000;
61  result += 4;
62  }
63 
64  return result;
65  };
66 
67  auto const result = digits10(val);
68  auto pos = result - 1;
69  while (val >= 10)
70  {
71  auto const q = val / 10;
72  auto const r = static_cast<char>(val % 10);
73  buffer[pos--] = static_cast<char>('0' + r);
74  val = q;
75  }
76 
77  *buffer = static_cast<char>(val + '0');
78  return buffer;
79  }
80  default:
81  {
82  ETL_ASSERT(false);
83  return buffer;
84  }
85  }
86 }
