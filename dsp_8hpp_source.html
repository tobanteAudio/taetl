106 template <typename T, int Z>
107 struct delay
108 {
109  constexpr delay() = default;
110  constexpr delay(T v)
111  {
112  for (auto& val : z_buffer_) { val = v; }
113  }
114 
115  constexpr auto operator()(T const& val)
116  {
117  z_buffer_[head_] = val;
118  if (++head_ > Z) { head_ = 0; }
119 
120  if (++tail_ > Z) { tail_ = 0; }
121  return z_buffer_[tail_];
122  };
123 
124 private:
125  using z_buffer_t = taetl::array<T, static_cast<size_t>(Z) + 1>;
126  typename z_buffer_t::size_type head_ = 0;
127  typename z_buffer_t::size_type tail_ = 0;
128  z_buffer_t z_buffer_ = {};
129 };
