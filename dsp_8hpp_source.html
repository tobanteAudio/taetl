110  {

112  }
113 
114  constexpr auto operator()(T const& val)
115  {
116  z_buffer_[head_] = val;
117  if (++head_ > Z) { head_ = 0; }
118 
119  if (++tail_ > Z) { tail_ = 0; }
120  return z_buffer_[tail_];
121  };
122 
123 private:
124  using z_buffer_t = etl::array<T, static_cast<size_t>(Z) + 1>;
125  typename z_buffer_t::size_type head_ = 0;
126  typename z_buffer_t::size_type tail_ = 0;
127  z_buffer_t z_buffer_ = {};
128 };
