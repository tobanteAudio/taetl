185 inline auto split_at_next_argument(etl::string_view str)
186  -> etl::pair<etl::string_view, etl::string_view>
187 {
188  using size_type = etl::string_view::size_type;
189 
190  auto const* res = etl::find(begin(str), end(str), token_begin);
191  if (res != end(str) && *etl::next(res) == token_end)
192  {
193  auto index = static_cast<size_type>(etl::distance(begin(str), res));
194  auto first = str.substr(0, index);
195  auto second = str.substr(index + 2);
196  return etl::make_pair(first, second);
197  }
198 
199  return etl::make_pair(str, etl::string_view {});
200 }

202 template <typename FormatContext>
203 auto format_escaped_sequences(::etl::string_view str, FormatContext& ctx) -> void
204 {
205  // Loop as long as escaped sequences are found.
206  auto const* first = begin(str);
207  while (true)
208  {
209  // Find open sequence {{
210  const auto* const open_first = ::etl::find(first, end(str), token_begin);
211  const auto* const open_sec = ::etl::next(open_first);
212  auto const escape_start = open_first != end(str) //
213  && open_sec != end(str) //
214  && *open_sec == token_begin;
215 
216  if (escape_start)
217  {
218  // Copy upto {{
219  detail::format_argument(etl::string_view(first, open_first), ctx);
220 
221  // Find sequence }}
222  auto const* close_first
223  = ::etl::find(::etl::next(open_sec), end(str), token_end);
224  auto const* close_sec = ::etl::next(close_first);
225  auto escape_close = close_first != end(str) //
226  && close_sec != end(str) //
227  && *close_sec == token_end;
228 
229  // Copy everything between {{ ... }}, but only one curly each.
230  if (escape_close)
231  {
232  detail::format_argument(etl::string_view(open_sec, close_first + 1), ctx);
233  first = close_first + 2;
234  }
235  else
236  {
237  assert(false && "No closing }} found");
238  return;
239  }
240  }
241  else
242  {
243  // No more escaped sequence found, copy rest.
244  detail::format_argument(etl::string_view(first, end(str)), ctx);
245  return;
246  }
247  }
248 }

258 template <typename OutputIt, typename... Args>
259 auto format_to(OutputIt out, etl::string_view fmt, Args const&... args) -> OutputIt
260 {
261  // TODO: Make more generic. What about other string types.
262  auto ctx = format_context<::etl::static_string<32>> {out};
263 
264  // Format leading text before the first argument.
265  auto const slices = detail::split_at_next_argument(fmt);
266  detail::format_escaped_sequences(slices.first, ctx);
267 
268  // Save rest of format string. Supress warning if format_to was called without
269  // arguments.
270  auto rest = slices.second;
271  ::etl::ignore_unused(rest);
272 
273  (
274  [&] {
275  // Format argument
276  detail::format_argument(args, ctx);
277 
278  // Split format text at next argument
279  auto const rest_slices = detail::split_at_next_argument(rest);
280  detail::format_escaped_sequences(rest_slices.first, ctx);
281 
282  // Save rest of format string for the next arguments
283  rest = rest_slices.second;
284  }(),
285  ...);
286 
287  // Anything left over after the last argument.
288  if (auto const trailing = detail::split_at_next_argument(rest);
289  !trailing.first.empty())
290  {
291  detail::format_escaped_sequences(trailing.first, ctx);
292  assert(trailing.second.empty());
293  }
294 
295  return ctx.out();
296 }

317 template <typename OutputIter, typename... Args>
318 auto format_to_n(OutputIter out, diff_t<OutputIter> n, ::etl::string_view fmt,
319  Args const&... args) -> format_to_n_result<OutputIter>
320 {
321  ::etl::ignore_unused(n);
322 
323  auto indices = ::etl::static_vector<::etl::size_t, sizeof...(args)> {};
324  auto result = format_to_n_result<OutputIter> {out, {}};
325 
326  auto write_char = [&result](auto ch) {
327  *result.out++ = ch;
328  result.size++;
329  };
330 
331  auto var_start = ::etl::size_t {};
332  for (decltype(fmt)::size_type i {}; i < fmt.size(); ++i)
333  {
334  auto ch = fmt[i];
335  if (ch == '{')
336  {
337  if (fmt[i + 1] == '{')
338  {
339  ++i;
340  write_char('{');
341  continue;
342  }
343 
344  var_start = i;
345  continue;
346  }
347 
348  if (ch == '}')
349  {
350  if (fmt[i + 1] == '}')
351  {
352  ++i;
353  write_char('}');
354  continue;
355  }
356 
357  indices.push_back(var_start);
358  write_char('0');
359  continue;
360  }
361 
362  write_char(ch);
363  }
364 
365  if (indices.size() > 0)
366  {
367  [[maybe_unused]] auto replace_char_at = [n](auto output, auto pos, char val) {
368  ::etl::ignore_unused(n);
369  // assert((long)pos < n);
370  output[pos] = val;
371  };
372 
373  [[maybe_unused]] typename decltype(indices)::size_type i {};
374  (replace_char_at(out, indices[i++], args), ...);
375  }
376 
377  return result;
378 }
