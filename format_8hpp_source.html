44 template <typename OutputIter, typename... Args>
45 auto format_to_n(OutputIter out, etl::ptrdiff_t n, etl::string_view fmt,
46  Args const&... args) -> format_to_n_result<OutputIter>
47 {
48  ::etl::ignore_unused(n);
49 
50  auto indices = etl::static_vector<etl::size_t, sizeof...(args)> {};
51  auto result = format_to_n_result<OutputIter> {out, {}};
52 
53  auto write_char = [&result](auto ch) {
54  *result.out++ = ch;
55  result.size++;
56  };
57 
58  auto var_start = etl::size_t {};
59  for (decltype(fmt)::size_type i {}; i < fmt.size(); ++i)
60  {
61  auto ch = fmt[i];
62  if (ch == '{')
63  {
64  if (fmt[i + 1] == '{')
65  {
66  ++i;
67  write_char('{');
68  continue;
69  }
70 
71  var_start = i;
72  continue;
73  }
74 
75  if (ch == '}')
76  {
77  if (fmt[i + 1] == '}')
78  {
79  ++i;
80  write_char('}');
81  continue;
82  }
83 
84  indices.push_back(var_start);
85  write_char('0');
86  continue;
87  }
88 
89  write_char(ch);
90  }
91 
92  if (indices.size() > 0)
93  {
94  [[maybe_unused]] auto replace_char_at = [n](auto output, auto pos, char val) {
95  ::etl::ignore_unused(n);
96  assert((long)pos < n);
97  output[pos] = val;
98  };
99 
100  [[maybe_unused]] typename decltype(indices)::size_type i {};
101  (replace_char_at(out, indices[i++], args), ...);
102  }
103 
104  return result;
105 }
