156  function_view(function_view const& other)
157  {
158  if (&other == this) { return; }
159  if (create_ptr_ != nullptr) { destroy_ptr_(storage_); }
160  if (other.create_ptr_ != nullptr)
161  {
162  invoke_ptr_ = other.invoke_ptr_;
163  create_ptr_ = other.create_ptr_;
164  destroy_ptr_ = other.destroy_ptr_;
165  create_ptr_(storage_, const_cast<etl::byte*>(other.storage_));
166  }
167  }
168

170  {

172  if (create_ptr_ != nullptr) { destroy_ptr_(storage_); }
173  if (other.create_ptr_ != nullptr)
174  {

176  create_ptr_ = other.create_ptr_;
177  destroy_ptr_ = other.destroy_ptr_;

182  }
