551 template <class Result, class... Arguments>
552 class function_view<Result(Arguments...)>
553 {
554 public:
555  using result_type = Result;
556 
557  function_view(function_view const& other)
558  {
559  if (&other == this) { return; }
560  if (create_ptr_ != nullptr) { destroy_ptr_(storage_); }
561  if (other.create_ptr_ != nullptr)
562  {
563  invoke_ptr_ = other.invoke_ptr_;
564  create_ptr_ = other.create_ptr_;
565  destroy_ptr_ = other.destroy_ptr_;
566  create_ptr_(storage_, const_cast<etl::byte*>(other.storage_));
567  }
568  }
569 
570  auto operator=(function_view const& other) noexcept -> function_view&
571  {
572  if (&other == this) { return *this; }
573  if (create_ptr_ != nullptr) { destroy_ptr_(storage_); }
574  if (other.create_ptr_ != nullptr)
575  {
576  invoke_ptr_ = other.invoke_ptr_;
577  create_ptr_ = other.create_ptr_;
578  destroy_ptr_ = other.destroy_ptr_;
579  create_ptr_(storage_, const_cast<etl::byte*>(other.storage_));
580  }
581 
582  return *this;
583  }
584 
585  ~function_view() noexcept { destroy_ptr_(storage_); }
586 
587  [[nodiscard]] auto operator()(Arguments&&... args) const -> result_type
588  {
589  return invoke_ptr_(const_cast<etl::byte*>(storage_),
590  etl::forward<Arguments>(args)...); // NOLINT
591  }
592 
593 protected:
594  template <typename Functor>
595  function_view(Functor f, etl::byte* storage)
596  : invoke_ptr_ {reinterpret_cast<invoke_pointer_t>(invoke<Functor>)}
597  , create_ptr_ {reinterpret_cast<create_pointer_t>(create<Functor>)}
598  , destroy_ptr_ {reinterpret_cast<destroy_pointer_t>(destroy<Functor>)}
599  , storage_ {storage}
600  {
601  create_ptr_(storage_, &f);
602  }
603 
604 private:
605  template <typename Functor>
606  static auto invoke(Functor* f, Arguments&&... args) -> Result
607  {
608  return (*f)(etl::forward<Arguments>(args)...);
609  }
610 
611  template <typename Functor>
612  static auto create(Functor* destination, Functor* source) -> void
613  {
614  new (destination) Functor(*source);
615  }
616 
617  template <typename Functor>
618  static auto destroy(Functor* f) -> void
619  {
620  f->~Functor();
621  }
622 
623  using invoke_pointer_t = Result (*)(void*, Arguments&&...);
624  using create_pointer_t = void (*)(void*, void*);
625  using destroy_pointer_t = void (*)(void*);
626 
627  invoke_pointer_t invoke_ptr_ = nullptr;
628  create_pointer_t create_ptr_ = nullptr;
629  destroy_pointer_t destroy_ptr_ = nullptr;
630 
631  etl::byte* storage_ = nullptr;
632 };
