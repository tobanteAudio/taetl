858 template <typename Result, typename... Arguments>
859 class function_view<Result(Arguments...)>
860 {
861 public:
862  using result_type = Result;
863 
864  function_view(function_view const& other)
865  {
866  if (&other == this) { return; }
867  if (create_ptr_ != nullptr) { destroy_ptr_(storage_); }
868  if (other.create_ptr_ != nullptr)
869  {
870  invoke_ptr_ = other.invoke_ptr_;
871  create_ptr_ = other.create_ptr_;
872  destroy_ptr_ = other.destroy_ptr_;
873  create_ptr_(storage_, const_cast<etl::byte*>(other.storage_));
874  }
875  }
876 
877  auto operator=(function_view const& other) noexcept -> function_view&
878  {
879  if (&other == this) { return *this; }
880  if (create_ptr_ != nullptr) { destroy_ptr_(storage_); }
881  if (other.create_ptr_ != nullptr)
882  {
883  invoke_ptr_ = other.invoke_ptr_;
884  create_ptr_ = other.create_ptr_;
885  destroy_ptr_ = other.destroy_ptr_;
886  create_ptr_(storage_, const_cast<etl::byte*>(other.storage_));
887  }
888 
889  return *this;
890  }
891 
892  ~function_view() noexcept { destroy_ptr_(storage_); }
893 
894  [[nodiscard]] auto operator()(Arguments&&... args) const -> result_type
895  {
896  return invoke_ptr_(const_cast<etl::byte*>(storage_),
897  etl::forward<Arguments>(args)...); // NOLINT
898  }
899 
900 protected:
901  template <typename Functor>
902  function_view(Functor f, etl::byte* storage)
903  : invoke_ptr_ {reinterpret_cast<invoke_pointer_t>(invoke<Functor>)}
904  , create_ptr_ {reinterpret_cast<create_pointer_t>(create<Functor>)}
905  , destroy_ptr_ {reinterpret_cast<destroy_pointer_t>(destroy<Functor>)}
906  , storage_ {storage}
907  {
908  create_ptr_(storage_, &f);
909  }
910 
911 private:
912  template <typename Functor>
913  static auto invoke(Functor* f, Arguments&&... args) -> Result
914  {
915  return (*f)(etl::forward<Arguments>(args)...);
916  }
917 
918  template <typename Functor>
919  static auto create(Functor* destination, Functor* source) -> void
920  {
921  new (destination) Functor(*source);
922  }
923 
924  template <typename Functor>
925  static auto destroy(Functor* f) -> void
926  {
927  f->~Functor();
928  }
929 
930  using invoke_pointer_t = Result (*)(void*, Arguments&&...);
931  using create_pointer_t = void (*)(void*, void*);
932  using destroy_pointer_t = void (*)(void*);
933 
934  invoke_pointer_t invoke_ptr_ = nullptr;
935  create_pointer_t create_ptr_ = nullptr;
936  destroy_pointer_t destroy_ptr_ = nullptr;
937 
938  etl::byte* storage_ = nullptr;
