280 class function_view<Result(Arguments...)>
281 {
282 public:
283  using result_type = Result;
284 
285  function_view(function_view const& other)
286  {
287  if (&other == this) { return; }
288  if (create_ptr_ != nullptr) { destroy_ptr_(storage_); }
289  if (other.create_ptr_ != nullptr)
290  {
291  invoke_ptr_ = other.invoke_ptr_;
292  create_ptr_ = other.create_ptr_;
293  destroy_ptr_ = other.destroy_ptr_;
294  create_ptr_(storage_, const_cast<etl::byte*>(other.storage_));
295  }
296  }
297 
298  auto operator=(function_view const& other) noexcept -> function_view&
299  {
300  if (&other == this) { return *this; }
301  if (create_ptr_ != nullptr) { destroy_ptr_(storage_); }
302  if (other.create_ptr_ != nullptr)
303  {
304  invoke_ptr_ = other.invoke_ptr_;
305  create_ptr_ = other.create_ptr_;
306  destroy_ptr_ = other.destroy_ptr_;
307  create_ptr_(storage_, const_cast<etl::byte*>(other.storage_));
308  }
309 
310  return *this;
311  }
312 
313  ~function_view() noexcept { destroy_ptr_(storage_); }
314 
315  [[nodiscard]] auto operator()(Arguments&&... args) const -> result_type
316  {
317  return invoke_ptr_(const_cast<etl::byte*>(storage_),
318  etl::forward<Arguments>(args)...); // NOLINT
319  }
320 
321 protected:
322  template <typename Functor>

324  : invoke_ptr_ {reinterpret_cast<invoke_pointer_t>(invoke<Functor>)}
325  , create_ptr_ {reinterpret_cast<create_pointer_t>(create<Functor>)}
