900 template <typename Result, typename... Arguments>
901 class function_view<Result(Arguments...)>
902 {
903  public:
904  using result_type = Result;
905 
906  function_view(function_view const& other)
907  {
908  if (&other == this) { return; }
909  if (create_ptr_ != nullptr) { destroy_ptr_(storage_); }
910  if (other.create_ptr_ != nullptr)
911  {
912  invoke_ptr_ = other.invoke_ptr_;
913  create_ptr_ = other.create_ptr_;
914  destroy_ptr_ = other.destroy_ptr_;
915  create_ptr_(storage_, const_cast<etl::byte*>(other.storage_));
916  }
917  }
918 
919  auto operator=(function_view const& other) noexcept -> function_view&
920  {
921  if (&other == this) { return *this; }
922  if (create_ptr_ != nullptr) { destroy_ptr_(storage_); }
923  if (other.create_ptr_ != nullptr)
924  {
925  invoke_ptr_ = other.invoke_ptr_;
926  create_ptr_ = other.create_ptr_;
927  destroy_ptr_ = other.destroy_ptr_;
928  create_ptr_(storage_, const_cast<etl::byte*>(other.storage_));
929  }
930 
931  return *this;
932  }
933 
934  ~function_view() noexcept { destroy_ptr_(storage_); }
935 
936  [[nodiscard]] auto operator()(Arguments&&... args) const -> result_type
937  {
938  return invoke_ptr_(const_cast<etl::byte*>(storage_),
939  etl::forward<Arguments>(args)...); // NOLINT
940  }
941 
942  protected:
943  template <typename Functor>
944  function_view(Functor f, etl::byte* storage)
945  : invoke_ptr_ {reinterpret_cast<invoke_pointer_t>(invoke<Functor>)}
946  , create_ptr_ {reinterpret_cast<create_pointer_t>(create<Functor>)}
947  , destroy_ptr_ {reinterpret_cast<destroy_pointer_t>(destroy<Functor>)}
948  , storage_ {storage}
949  {
950  create_ptr_(storage_, &f);
951  }
952 
953  private:
954  template <typename Functor>
955  static auto invoke(Functor* f, Arguments&&... args) -> Result
956  {
957  return (*f)(etl::forward<Arguments>(args)...);
958  }
959 
960  template <typename Functor>
961  static auto create(Functor* destination, Functor* source) -> void
962  {
963  new (destination) Functor(*source);
964  }
965 
966  template <typename Functor>
967  static auto destroy(Functor* f) -> void
968  {
969  f->~Functor();
970  }
971 
972  using invoke_pointer_t = Result (*)(void*, Arguments&&...);
973  using create_pointer_t = void (*)(void*, void*);
974  using destroy_pointer_t = void (*)(void*);
975 
976  invoke_pointer_t invoke_ptr_ = nullptr;
977  create_pointer_t create_ptr_ = nullptr;
978  destroy_pointer_t destroy_ptr_ = nullptr;
979 
980  etl::byte* storage_ = nullptr;
981 };
