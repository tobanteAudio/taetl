711 template <class Result, class... Arguments>
712 class function_view<Result(Arguments...)>
713 {
714 public:
715  using result_type = Result;
716 
717  function_view(function_view const& other)
718  {
719  if (&other == this) { return; }
720  if (create_ptr_ != nullptr) { destroy_ptr_(storage_); }
721  if (other.create_ptr_ != nullptr)
722  {
723  invoke_ptr_ = other.invoke_ptr_;
724  create_ptr_ = other.create_ptr_;
725  destroy_ptr_ = other.destroy_ptr_;
726  create_ptr_(storage_, const_cast<etl::byte*>(other.storage_));
727  }
728  }
729 
730  auto operator=(function_view const& other) noexcept -> function_view&
731  {
732  if (&other == this) { return *this; }
733  if (create_ptr_ != nullptr) { destroy_ptr_(storage_); }
734  if (other.create_ptr_ != nullptr)
735  {
736  invoke_ptr_ = other.invoke_ptr_;
737  create_ptr_ = other.create_ptr_;
738  destroy_ptr_ = other.destroy_ptr_;
739  create_ptr_(storage_, const_cast<etl::byte*>(other.storage_));
740  }
741 
742  return *this;
743  }
744 
745  ~function_view() noexcept { destroy_ptr_(storage_); }
746 
747  [[nodiscard]] auto operator()(Arguments&&... args) const -> result_type
748  {
749  return invoke_ptr_(const_cast<etl::byte*>(storage_),
750  etl::forward<Arguments>(args)...); // NOLINT
751  }
752 
753 protected:
754  template <typename Functor>
755  function_view(Functor f, etl::byte* storage)
756  : invoke_ptr_ {reinterpret_cast<invoke_pointer_t>(invoke<Functor>)}
757  , create_ptr_ {reinterpret_cast<create_pointer_t>(create<Functor>)}
758  , destroy_ptr_ {reinterpret_cast<destroy_pointer_t>(destroy<Functor>)}
759  , storage_ {storage}
760  {
761  create_ptr_(storage_, &f);
762  }
763 
764 private:
765  template <typename Functor>
766  static auto invoke(Functor* f, Arguments&&... args) -> Result
767  {
768  return (*f)(etl::forward<Arguments>(args)...);
769  }
770 
771  template <typename Functor>
772  static auto create(Functor* destination, Functor* source) -> void
773  {
774  new (destination) Functor(*source);
775  }
776 
777  template <typename Functor>
778  static auto destroy(Functor* f) -> void
779  {
780  f->~Functor();
781  }
782 
783  using invoke_pointer_t = Result (*)(void*, Arguments&&...);
784  using create_pointer_t = void (*)(void*, void*);
785  using destroy_pointer_t = void (*)(void*);
786 
787  invoke_pointer_t invoke_ptr_ = nullptr;
788  create_pointer_t create_ptr_ = nullptr;
789  destroy_pointer_t destroy_ptr_ = nullptr;
790 
791  etl::byte* storage_ = nullptr;
792 };

822 {

825  : first_(f), last_(l), predicate_(p)

828 
829  template <typename ForwardIter2>
830  auto operator()(ForwardIter2 f, ForwardIter2 l) const
831  -> etl::pair<ForwardIter2, ForwardIter2>
832  {
833  if (auto i = ::etl::detail::search_impl(f, l, first_, last_, predicate_); i != l)
834  {
835  auto j = ::etl::next(i, etl::distance(first_, last_));
836  return etl::pair<ForwardIter2, ForwardIter2> {i, j};
837  }
838

840  }
