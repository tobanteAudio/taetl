313 template <class It, class Distance>
314 constexpr auto advance(It& it, Distance n) -> void
315 {
316  using category = typename etl::iterator_traits<It>::iterator_category;
317  static_assert(etl::is_base_of_v<etl::input_iterator_tag, category>);
318 
319  auto dist = typename etl::iterator_traits<It>::difference_type(n);
320  if constexpr (etl::is_base_of_v<etl::random_access_iterator_tag, category>)
321  it += dist;
322  else
323  {
324  while (dist > 0)
325  {
326  --dist;
327  ++it;
328  }
329  if constexpr (etl::is_base_of_v<etl::bidirectional_iterator_tag,
330  category>)
331  {
332  while (dist < 0)
333  {
334  ++dist;
335  --it;
336  }
337  }
338  }
339 }

349 {

351  static_assert(etl::is_base_of_v<etl::input_iterator_tag, category>);
352 
353  if constexpr (etl::is_base_of_v<etl::random_access_iterator_tag, category>)
354  return last - first;
355  else
356  {
357  typename etl::iterator_traits<It>::difference_type result = 0;
358  while (first != last)
