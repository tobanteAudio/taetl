135 template <typename It, typename Distance>
136 constexpr auto advance(It& it, Distance n) -> void
137 {
138  using category = typename etl::iterator_traits<It>::iterator_category;
139  static_assert(etl::is_base_of_v<etl::input_iterator_tag, category>);
140 
141  auto dist = typename etl::iterator_traits<It>::difference_type(n);
142  if constexpr (etl::is_base_of_v<etl::random_access_iterator_tag, category>)
143  { it += dist; }
144  else
145  {
146  while (dist > 0)
147  {
148  --dist;
149  ++it;
150  }
151  if constexpr (etl::is_base_of_v<etl::bidirectional_iterator_tag, category>)
152  {
153  while (dist < 0)
154  {
155  ++dist;
156  --it;
157  }
158  }
159  }
160 }

167 template <typename It>
168 constexpr auto distance(It first, It last) ->
169  typename etl::iterator_traits<It>::difference_type
170 {
171  using category = typename etl::iterator_traits<It>::iterator_category;
172  static_assert(etl::is_base_of_v<etl::input_iterator_tag, category>);
173 
174  if constexpr (etl::is_base_of_v<etl::random_access_iterator_tag, category>)
175  { return last - first; }
176  else
177  {
178  typename etl::iterator_traits<It>::difference_type result = 0;
179  while (first != last)
180  {
181  ++first;
182  ++result;
183  }
184  return result;
185  }
186 }
