311 template <class It, class Distance>
312 constexpr auto advance(It& it, Distance n) -> void
313 {
314  using category = typename etl::iterator_traits<It>::iterator_category;
315  static_assert(etl::is_base_of_v<etl::input_iterator_tag, category>);
316 
317  auto dist = typename etl::iterator_traits<It>::difference_type(n);
318  if constexpr (etl::is_base_of_v<etl::random_access_iterator_tag, category>)
319  { it += dist; }
320  else
321  {
322  while (dist > 0)
323  {
324  --dist;
325  ++it;
326  }
327  if constexpr (etl::is_base_of_v<etl::bidirectional_iterator_tag, category>)
328  {
329  while (dist < 0)
330  {
331  ++dist;
332  --it;
333  }
334  }
335  }
336 }

343 template <class It>
344 constexpr auto distance(It first, It last) ->
345  typename etl::iterator_traits<It>::difference_type
346 {
347  using category = typename etl::iterator_traits<It>::iterator_category;
348  static_assert(etl::is_base_of_v<etl::input_iterator_tag, category>);
349 
350  if constexpr (etl::is_base_of_v<etl::random_access_iterator_tag, category>)
351  { return last - first; }
352  else
353  {
354  typename etl::iterator_traits<It>::difference_type result = 0;
355  while (first != last)
356  {
357  ++first;
358  ++result;
359  }
360  return result;
361  }
362 }
