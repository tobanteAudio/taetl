315 template <class It, class Distance>
316 constexpr auto advance(It& it, Distance n) -> void
317 {
318  using category = typename etl::iterator_traits<It>::iterator_category;
319  static_assert(etl::is_base_of_v<etl::input_iterator_tag, category>);
320 
321  auto dist = typename etl::iterator_traits<It>::difference_type(n);
322  if constexpr (etl::is_base_of_v<etl::random_access_iterator_tag, category>)
323  it += dist;
324  else
325  {
326  while (dist > 0)
327  {
328  --dist;
329  ++it;
330  }
331  if constexpr (etl::is_base_of_v<etl::bidirectional_iterator_tag,
332  category>)
333  {
334  while (dist < 0)
335  {
336  ++dist;
337  --it;
338  }
339  }
340  }
341 }

348 template <class It>
349 constexpr auto distance(It first, It last) ->
350  typename etl::iterator_traits<It>::difference_type
351 {
352  using category = typename etl::iterator_traits<It>::iterator_category;
353  static_assert(etl::is_base_of_v<etl::input_iterator_tag, category>);
354 
355  if constexpr (etl::is_base_of_v<etl::random_access_iterator_tag, category>)
356  return last - first;
357  else
358  {
359  typename etl::iterator_traits<It>::difference_type result = 0;
360  while (first != last)
361  {
362  ++first;
363  ++result;
364  }
365  return result;
366  }
367 }
