250  template <typename... Args>
251  constexpr auto emplace(Args&&... args) -> etl::pair<iterator, bool>
252  {
253  // Return if no capacity is left.
254  if (size_ == capacity_) { return {end(), false}; }
255 
256  // Construct value_type inplace at the end of the internal array.
257  auto* const addr = reinterpret_cast<void*>(&data_[size_]);
258  auto* obj = ::new (addr) value_type {etl::forward<Args>(args)...};
259 
260  // Check if the key from the newly created object has already existed.
261  auto* key_existed = find_if(
262  begin(), end(), [&](auto const& item) { return item.first == obj->first; });
263 
264  // If so, return its iterator and false for insertion.
265  if (key_existed != end())
266  {
267  obj->~value_type();
268  return {key_existed, false};
269  }
270 
271  // Key has not existed before. Array needs to be sorted.
272  size_++;
273  return {obj, true};
274  }
