262  template <typename... Args>
263  constexpr auto emplace(Args&&... args) -> etl::pair<iterator, bool>
264  {
265  // Return if no capacity is left.
266  if (size_ == capacity_) { return {end(), false}; }
267 
268  // Construct value_type inplace at the end of the internal array.
269  auto* const addr = reinterpret_cast<void*>(&data_[size_]);
270  auto* obj = ::new (addr) value_type {etl::forward<Args>(args)...};
271 
272  // Check if the key from the newly created object has already existed.
273  auto* key_existed = find_if(begin(), end(), [&](auto const& item) {
274  return item.first == obj->first;
275  });
276 
277  // If so, return its iterator and false for insertion.
278  if (key_existed != end())
279  {
280  obj->~value_type();
281  return {key_existed, false};
282  }
283 
284  // Key has not existed before. Array needs to be sorted.
285  size_++;
286  return {obj, true};
287  }
