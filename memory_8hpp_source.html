298  auto mask = storage_type {0};
299  for (storage_type i = 0; i < pointer_bits; ++i)
300  { mask |= 1UL << (i + int_bits); }
301  return mask;
302  }();
303 
304  static constexpr storage_type integer_mask = [] {
305  auto mask = storage_type {0};
306  for (storage_type i = 0; i < int_bits; ++i) { mask |= 1UL << i; }
307  return mask;
308  }();

310  auto internal_store_ptr(pointer ptr) noexcept -> void
311  {
312  auto const tmp = reinterpret_cast<uintptr_t>(ptr);
313  if constexpr (int_bits > 0)
314  {
315  assert(tmp <= (1UL << pointer_bits));
316  storage_ |= static_cast<storage_type>(tmp) << pointer_bits;
317  return;
318  }
319  storage_ |= static_cast<storage_type>(tmp);
320  }
321 
322  auto internal_store_int(int_type integer) noexcept -> void
323  {
324  assert(integer <= (1UL << int_bits));
325  storage_ &= (~integer_mask);
326  storage_ |= (integer & integer_mask);
327  }
328 
329  [[nodiscard]] auto internal_load_ptr() const noexcept -> pointer
330  {
331  if constexpr (int_bits > 0)
332  {
333  auto const tmp = storage_ & pointer_mask;
334  return reinterpret_cast<pointer>(tmp >> pointer_bits);
335  }
336  return reinterpret_cast<pointer>(storage_ & pointer_mask);
337  }
338 
339  [[nodiscard]] auto internal_load_int() const noexcept -> int_type
340  {

342  }
