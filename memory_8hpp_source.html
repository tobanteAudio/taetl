420  auto mask = storage_type {0};
421  for (storage_type i = 0; i < pointer_bits; ++i)
422  { mask |= 1UL << (i + int_bits); }
423  return mask;
424  }();
425 
426  static constexpr storage_type integer_mask = [] {
427  auto mask = storage_type {0};
428  for (storage_type i = 0; i < int_bits; ++i) { mask |= 1UL << i; }
429  return mask;
430  }();

432  auto internal_store_ptr(pointer ptr) noexcept -> void
433  {
434  auto const tmp = reinterpret_cast<uintptr_t>(ptr);
435  if constexpr (int_bits > 0)
436  {
437  assert(tmp <= (1UL << pointer_bits));
438  storage_ |= static_cast<storage_type>(tmp) << pointer_bits;
439  return;
440  }
441  storage_ |= static_cast<storage_type>(tmp);
442  }
443 
444  auto internal_store_int(int_type integer) noexcept -> void
445  {
446  assert(integer <= (1UL << int_bits));
447  storage_ &= (~integer_mask);
448  storage_ |= (integer & integer_mask);
449  }
450 
451  [[nodiscard]] auto internal_load_ptr() const noexcept -> pointer
452  {
453  if constexpr (int_bits > 0)
454  {
455  auto const tmp = storage_ & pointer_mask;
456  return reinterpret_cast<pointer>(tmp >> pointer_bits);
457  }
458  return reinterpret_cast<pointer>(storage_ & pointer_mask);
459  }
460 
461  [[nodiscard]] auto internal_load_int() const noexcept -> int_type
462  {
463  return storage_ & integer_mask;
464  }
465 
466  storage_type storage_ = 0;
467 };
