266 constexpr auto midpoint(Int a, Int b) noexcept -> Int
267 {
268  using U = make_unsigned_t<Int>;
269 
270  auto sign = 1;
271  auto m = static_cast<U>(a);
272  auto M = static_cast<U>(b);

294 template <typename Float, TAETL_REQUIRES_(is_floating_point_v<Float>)>
295 constexpr auto midpoint(Float a, Float b) noexcept -> Float
296 {
297  auto const lo = numeric_limits<Float>::min() * 2;
298  auto const hi = numeric_limits<Float>::max() / 2;
299 
300  if (etl::abs(a) <= hi && etl::abs(b) <= hi) { return (a + b) / 2; }
301 
302  if (etl::abs(a) < lo) { return a + b / 2; }
303 
304  if (etl::abs(b) < lo) { return a / 2 + b; }
305
