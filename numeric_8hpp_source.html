303 template <typename Float, TAETL_REQUIRES_(is_floating_point_v<Float>)>
304 constexpr auto midpoint(Float a, Float b) noexcept -> Float
305 {
306  auto const lo = numeric_limits<Float>::min() * 2;
307  auto const hi = numeric_limits<Float>::max() / 2;
308 
309  if (etl::abs(a) <= hi && etl::abs(b) <= hi) { return (a + b) / 2; }
