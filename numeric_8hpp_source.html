97 template <typename Integer>
98 constexpr auto midpoint(Integer a, Integer b) noexcept
99  -> etl::enable_if_t<etl::is_integral_v<Integer>, Integer>
100 {
101  using U = etl::make_unsigned_t<Integer>;
102  int sign = 1;
103  auto m = static_cast<U>(a);
104  auto M = static_cast<U>(b);
105  if (a > b)
106  {
107  sign = -1;
108  m = static_cast<U>(b);
109  M = static_cast<U>(a);
110  }
111  return static_cast<Integer>(
112  a + static_cast<Integer>(sign * static_cast<Integer>(U(M - m) >> 1)));
113 }

115 template <typename Float>
116 constexpr auto midpoint(Float a, Float b) noexcept
117  -> etl::enable_if_t<etl::is_floating_point_v<Float>, Float>
118 {
119  auto const lo = etl::numeric_limits<Float>::min() * 2;
120  auto const hi = etl::numeric_limits<Float>::max() / 2;
121 
122  if (etl::abs(a) <= hi && etl::abs(b) <= hi) { return (a + b) / 2; }
123 
124  if (etl::abs(a) < lo) { return a + b / 2; }
125 
126  if (etl::abs(b) < lo) { return a / 2 + b; }
127 
128  return a / 2 + b / 2;
129 }
