533  constexpr auto swap(optional& other) noexcept(
534  etl::is_nothrow_move_constructible_v<value_type>&&
535  etl::is_nothrow_swappable_v<value_type>) -> void
536  {
537  // If neither *this nor other contain a value, the function has no effect.
538 
539  // If both *this and other contain values, the contained values are
540  // exchanged
541  if (this->has_value() == other.has_value())
542  {
543  using etl::swap;
544  if (this->has_value()) { swap(this->get(), other.get()); }
545  return;
546  }
547 
548  // If only one of *this and other contains a value (let's call this object
549  // in and the other un), the contained value of un is direct-initialized
550  // from etl::move(*in), followed by destruction of the contained value of in
551  // as if by in->T::~T(). After this call, in does not contain a value; un
552  // contains a value.
553  if (this->has_value())
554  {
555  other.construct(etl::move(this->get()));
556  reset();
557  return;
558  }
559 
560  this->construct(etl::move(other.get()));
561  other.reset();
562  }
