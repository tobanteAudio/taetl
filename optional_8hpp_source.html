146  void construct(Args&&... args)
147  {

150  this->has_value_ = true;

153  template <class T>

155  {

159  template <class T>
160 
161  void assign_from(T&& opt)
162  {
163  if (this->has_value_ == opt.has_value())
164  {
165  if (this->has_value_) this->value_ = etl::forward<T>(opt).get();
166  }
167  else
168  {
169  if (this->has_value_)
170  this->reset();
171  else
172  construct(etl::forward<T>(opt).get());
173  }
174  }
