122  constexpr auto operator=(optional const& other) -> optional&
123  {
124  // Self assignment
125  if (this == &other) { return *this; }
126  // If both *this and other do not contain a value, the function has no
127  // effect.
128  if (!this->has_value() && !other.has_value()) { return *this; }
129 
130  // If *this contains a value, but other does not, then the contained value
131  // is destroyed by calling its destructor. *this does not contain a value
132  // after the call.
133  if (this->has_value() && !other.has_value())
134  {
135  this->reset();
136  return *this;
137  }
138 
139  // If other contains a value, then depending on whether *this contains a
140  // value, the contained value is either direct-initialized or assigned from
141  // *other (2) or etl::move(*other) (3). Note that a moved-from optional
142  // still contains a value.
143  if (other.has_value())
144  {
145  if (this->has_value()) { (*this) = *other; }
146  else
147  {
148  (*this) = etl::move(*other);
149  }
150  return *this;
151  }
152 
153  return *this;
154  }
