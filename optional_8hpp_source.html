497  constexpr auto
498  swap(optional& other) noexcept(etl::is_nothrow_move_constructible_v<value_type>&&
499  etl::is_nothrow_swappable_v<value_type>) -> void
500  {
501  // If neither *this nor other contain a value, the function has no effect.
502 
503  // If both *this and other contain values, the contained values are exchanged
504  if (this->has_value() == other.has_value())
505  {
506  using etl::swap;
507  if (this->has_value()) { swap(this->get(), other.get()); }
508  return;
509  }
510 
511  // If only one of *this and other contains a value (let's call this object in
512  // and the other un), the contained value of un is direct-initialized from
513  // etl::move(*in), followed by destruction of the contained value of in as if
514  // by in->T::~T(). After this call, in does not contain a value; un contains a
515  // value.
516  if (this->has_value())
517  {
518  other.construct(etl::move(this->get()));
519  reset();
520  return;
521  }
522 
523  this->construct(etl::move(other.get()));
524  other.reset();
525  }
526 
530  using base_type::get;
