214  auto insert(value_type&& value) -> etl::pair<iterator, bool>
215  {
216  if (!full())
217  {
218  auto p = etl::lower_bound(begin(), end(), value);
219  if (p == end() || *(p) != value)
220  {
221  data_[size_++] = etl::move(value);
222  auto pos = etl::rotate(p, end() - 1, end());
223  return etl::make_pair(pos, true);
224  }
225  }
226 
227  return etl::pair<iterator, bool>(nullptr, false);
228  }
