289  constexpr auto insert(value_type&& value)
290  -> enable_if_t<is_move_constructible_v<value_type>, etl::pair<iterator, bool>>
291  {
292  if (!full())
293  {
294  auto* p = lower_bound(begin(), end(), value, key_compare {});
295  if (p == end() || *(p) != value)
296  {
297  data_[size_++] = etl::move(value);
298  auto* pos = etl::rotate(p, end() - 1, end());
299  return etl::make_pair(pos, true);
300  }
301  }
302 
303  return etl::pair<iterator, bool>(nullptr, false);
304  }
