280  constexpr auto insert(value_type&& value) -> etl::pair<iterator, bool>
281  {
282  if (!full())
283  {
284  auto p = etl::lower_bound(begin(), end(), value, key_compare {});
285  if (p == end() || *(p) != value)
286  {
287  data_[size_++] = etl::move(value);
