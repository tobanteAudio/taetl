962  [[nodiscard]] constexpr auto compare_impl(const_pointer lhs, size_type lhs_size,
963  const_pointer rhs,
964  size_type rhs_size) const noexcept -> int
965  {
966  auto const min_size = etl::min(lhs_size, rhs_size);
967  auto const result = traits_type::compare(lhs, rhs, min_size);
968  if (result != 0) { return result; }
969  if (lhs_size < rhs_size) { return -1; }
970  if (lhs_size > rhs_size) { return 1; }
971  return 0;
