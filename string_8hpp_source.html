615  [[nodiscard]] constexpr auto compare_impl(const_pointer lhs, size_type lhs_size,
616  const_pointer rhs,
617  size_type rhs_size) const noexcept -> int
618  {
619  auto const min_size = etl::min(lhs_size, rhs_size);
620  auto const result = traits_type::compare(lhs, rhs, min_size);
621  if (result != 0) { return result; }
622  if (lhs_size < rhs_size) { return -1; }
623  if (lhs_size > rhs_size) { return 1; }
624  return 0;
625  }
