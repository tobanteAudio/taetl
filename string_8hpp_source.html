516  [[nodiscard]] constexpr auto compare_impl(const value_type* lhs, size_type lhs_size,
517  const value_type* rhs,
518  size_type rhs_size) const noexcept -> int
519  {
520  auto const min_size = etl::min(lhs_size, rhs_size);
521  auto const result = traits_type::compare(lhs, rhs, min_size);
522  if (result != 0) { return result; }
523  if (lhs_size < rhs_size) { return -1; }
524  if (lhs_size > rhs_size) { return 1; }
525  return 0;
526  }
