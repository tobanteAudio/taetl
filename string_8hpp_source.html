977  [[nodiscard]] constexpr auto compare_impl(const_pointer lhs, size_type lhs_size,
978  const_pointer rhs,
979  size_type rhs_size) const noexcept -> int
980  {
981  auto const min_size = etl::min(lhs_size, rhs_size);
982  auto const result = traits_type::compare(lhs, rhs, min_size);
983  if (result != 0) { return result; }
984  if (lhs_size < rhs_size) { return -1; }
985  if (lhs_size > rhs_size) { return 1; }
986  return 0;
