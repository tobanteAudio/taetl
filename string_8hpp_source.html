849  [[nodiscard]] constexpr auto compare_impl(const_pointer lhs, size_type lhs_size,
850  const_pointer rhs,
851  size_type rhs_size) const noexcept -> int
852  {
853  auto const min_size = etl::min(lhs_size, rhs_size);
854  auto const result = traits_type::compare(lhs, rhs, min_size);
855  if (result != 0) { return result; }
856  if (lhs_size < rhs_size) { return -1; }
857  if (lhs_size > rhs_size) { return 1; }
858  return 0;
