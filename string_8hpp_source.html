905  [[nodiscard]] constexpr auto compare_impl(const_pointer lhs, size_type lhs_size,
906  const_pointer rhs,
907  size_type rhs_size) const noexcept -> int
908  {
909  auto const min_size = etl::min(lhs_size, rhs_size);
910  auto const result = traits_type::compare(lhs, rhs, min_size);
911  if (result != 0) { return result; }
912  if (lhs_size < rhs_size) { return -1; }
913  if (lhs_size > rhs_size) { return 1; }
914  return 0;
