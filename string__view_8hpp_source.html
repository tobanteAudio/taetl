412  constexpr auto find(basic_string_view v, size_type pos = 0) const noexcept
413  -> size_type
414  {
415  if (v.size() > size() - pos) { return npos; }
416 
417  for (size_type outerIdx = pos; outerIdx < size(); ++outerIdx)
418  {
419  if (at(outerIdx) == v.front())
420  {
421  auto found = [&] {
422  for (size_type innerIdx = 0; innerIdx < v.size();
423  ++innerIdx)
424  {
425  auto offset = outerIdx + innerIdx;
426  if (at(offset) != v.at(innerIdx)) { return false; }
427  }
428 
429  return true;
430  }();
431 
432  if (found) { return outerIdx; }
433  }
434  }
435 
436  return npos;
437  }

484  constexpr auto rfind(basic_string_view v,
485  size_type pos = npos) const noexcept -> size_type
486  {
487  auto const offset = etl::clamp<size_type>(pos, 0, size());
488  if (v.size() > size()) { return npos; }
489 
490  size_type outer = offset;
491  do {
492  if (at(outer) == v.front())
493  {
494  auto found = [&] {
495  for (size_type inner = 0; inner < v.size(); ++inner)
496  {
497  if (at(outer + inner) != v.at(inner)) { return false; }
498  }
499 
500  return true;
501  }();
502 
503  if (found) { return outer; }
504  }
505  } while (outer-- != 0);
506 
507  return npos;
508  }
