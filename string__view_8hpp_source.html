476  [[nodiscard]] constexpr auto find(basic_string_view v,
477  size_type pos = 0) const noexcept
478  -> size_type
479  {
480  if (v.size() > size() - pos) { return npos; }
481 
482  for (size_type outerIdx = pos; outerIdx < size(); ++outerIdx)
483  {
484  if (unsafe_at(outerIdx) == v.front())
485  {
486  auto found = [&] {
487  for (size_type innerIdx = 0; innerIdx < v.size(); ++innerIdx)
488  {
489  auto offset = outerIdx + innerIdx;
490  if (unsafe_at(offset) != v[innerIdx]) { return false; }
491  }
492 
493  return true;
494  }();
495 
496  if (found) { return outerIdx; }
497  }
498  }
499 
500  return npos;
501  }

550  size_type pos = npos) const noexcept
551  -> size_type
