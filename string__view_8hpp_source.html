391  [[nodiscard]] constexpr auto find(basic_string_view v,
392  size_type pos = 0) const noexcept -> size_type
393  {
394  if (v.size() > size() - pos) { return npos; }
395 
396  for (size_type outerIdx = pos; outerIdx < size(); ++outerIdx)
397  {
398  if (at(outerIdx) == v.front())
399  {
400  auto found = [&] {
401  for (size_type innerIdx = 0; innerIdx < v.size(); ++innerIdx)
402  {
403  auto offset = outerIdx + innerIdx;
404  if (at(offset) != v.at(innerIdx)) { return false; }
405  }
406 
407  return true;
408  }();
409 
410  if (found) { return outerIdx; }
411  }
412  }
413 
414  return npos;
415  }

462  [[nodiscard]] constexpr auto rfind(basic_string_view v,
463  size_type pos = npos) const noexcept -> size_type
464  {
465  auto const offset = etl::clamp<size_type>(pos, 0, size());
466  if (v.size() > size()) { return npos; }
467 
468  size_type outer = offset;
469  do {
470  if (at(outer) == v.front())
471  {
472  auto found = [&] {
473  for (size_type inner = 0; inner < v.size(); ++inner)
474  {
475  if (at(outer + inner) != v.at(inner)) { return false; }
476  }
477 
478  return true;
479  }();
480 
481  if (found) { return outer; }
482  }
483  } while (outer-- != 0);
484 
485  return npos;
486  }
