453  [[nodiscard]] constexpr auto find(basic_string_view v,
454  size_type pos = 0) const noexcept -> size_type
455  {
456  if (v.size() > size() - pos) { return npos; }
457 
458  for (size_type outerIdx = pos; outerIdx < size(); ++outerIdx)
459  {
460  if (unsafe_at(outerIdx) == v.front())
461  {
462  auto found = [&] {
463  for (size_type innerIdx = 0; innerIdx < v.size(); ++innerIdx)
464  {
465  auto offset = outerIdx + innerIdx;
466  if (unsafe_at(offset) != v[innerIdx]) { return false; }
467  }
468 
469  return true;
470  }();
471 
472  if (found) { return outerIdx; }
473  }
474  }
475 
476  return npos;
477  }

524  [[nodiscard]] constexpr auto rfind(basic_string_view v,
525  size_type pos = npos) const noexcept -> size_type
526  {
527  auto const offset = etl::clamp<size_type>(pos, 0, size());
528  if (v.size() > size()) { return npos; }
