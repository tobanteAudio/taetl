463  [[nodiscard]] constexpr auto find(basic_string_view v,
464  size_type pos = 0) const noexcept -> size_type
465  {
466  if (v.size() > size() - pos) { return npos; }
467 
468  for (size_type outerIdx = pos; outerIdx < size(); ++outerIdx)
469  {
470  if (at(outerIdx) == v.front())
471  {
472  auto found = [&] {
473  for (size_type innerIdx = 0; innerIdx < v.size(); ++innerIdx)
474  {
475  auto offset = outerIdx + innerIdx;
476  if (at(offset) != v.at(innerIdx)) { return false; }
477  }
478 
479  return true;
480  }();
481 
482  if (found) { return outerIdx; }
483  }
484  }
485 
486  return npos;
487  }

534  [[nodiscard]] constexpr auto rfind(basic_string_view v,
535  size_type pos = npos) const noexcept -> size_type
536  {
537  auto const offset = etl::clamp<size_type>(pos, 0, size());
538  if (v.size() > size()) { return npos; }
539 
540  size_type outer = offset;
541  do {

548  }
