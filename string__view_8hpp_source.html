409  [[nodiscard]] constexpr auto find(basic_string_view v,
410  size_type pos = 0) const noexcept -> size_type
411  {
412  if (v.size() > size() - pos) { return npos; }
413 
414  for (size_type outerIdx = pos; outerIdx < size(); ++outerIdx)
415  {
416  if (at(outerIdx) == v.front())
417  {
418  auto found = [&] {
419  for (size_type innerIdx = 0; innerIdx < v.size(); ++innerIdx)
420  {
421  auto offset = outerIdx + innerIdx;
422  if (at(offset) != v.at(innerIdx)) { return false; }
423  }
424 
425  return true;
426  }();
427 
428  if (found) { return outerIdx; }
429  }
430  }
431 
432  return npos;
433  }

480  [[nodiscard]] constexpr auto rfind(basic_string_view v,
481  size_type pos = npos) const noexcept -> size_type
482  {
483  auto const offset = etl::clamp<size_type>(pos, 0, size());
484  if (v.size() > size()) { return npos; }
485 
486  size_type outer = offset;
487  do {
488  if (at(outer) == v.front())
489  {
490  auto found = [&] {
491  for (size_type inner = 0; inner < v.size(); ++inner)
492  {
493  if (at(outer + inner) != v.at(inner)) { return false; }
494  }
495 
496  return true;
497  }();
498 
499  if (found) { return outer; }
500  }
501  } while (outer-- != 0);
502 
503  return npos;
