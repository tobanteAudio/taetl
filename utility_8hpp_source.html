143 template <class T, class U>
144 [[nodiscard]] constexpr auto cmp_equal(T t, U u) noexcept
145  -> enable_if_t<detail::is_integer_and_not_char<T>::value
146  && detail::is_integer_and_not_char<U>::value,
147  bool>
148 {
149  using UT = etl::make_unsigned_t<T>;
150  using UU = etl::make_unsigned_t<U>;
151 
152  if constexpr (etl::is_signed_v<T> == etl::is_signed_v<U>) { return t == u; }
153  else if constexpr (etl::is_signed_v<T>)
154  {
155  return t < 0 ? false : UT(t) == u;
156  }
157  else
158  {
159  return u < 0 ? false : t == UU(u);
160  }
161 }
