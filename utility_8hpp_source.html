145 template <class T, class U>
146 [[nodiscard]] constexpr auto cmp_equal(T t, U u) noexcept
147  -> enable_if_t<detail::is_integer_and_not_char<T>::value
148  && detail::is_integer_and_not_char<U>::value,
149  bool>
150 {
151  using UT = etl::make_unsigned_t<T>;
152  using UU = etl::make_unsigned_t<U>;
153 
154  if constexpr (etl::is_signed_v<T> == etl::is_signed_v<U>) { return t == u; }
155  else if constexpr (etl::is_signed_v<T>)
156  {
157  return t < 0 ? false : UT(t) == u;
158  }
159  else
160  {
161  return u < 0 ? false : t == UU(u);
162  }
163 }
