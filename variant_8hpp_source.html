305  constexpr auto operator=(variant const& rhs) -> variant&
306  {
307  // Self assignment or Both Valueless
308  if ((this == &rhs) || (valueless_by_exception() && rhs.valueless_by_exception()))
309  {
310  // Do nothing
311  return *this;
312  }
313 
314  // RHS Valueless
315  if (!valueless_by_exception() && rhs.valueless_by_exception())
316  {
317  data_.destruct(union_index_);
318  return *this;
319  }
320 
321  // Same type
322  if (index() && rhs.index())
323  {
324  data_ = rhs.data_;
325  return *this;
326  }
327 
328  return *this;
329  }
