164  template <typename T>
165  void construct(T&& t, union_index_type& union_index)
166  {
167  tail.construct(etl::forward<T>(t), union_index);
168  ++union_index;

171  void destruct(union_index_type union_index)
172  {
173  if (union_index == 0)
174  {
175  static_cast<Head*>(static_cast<void*>(&data))->~Head();
176  return;
177  }
178 
179  tail.destruct(union_index - 1);
